    saveGraphToRedux() {
        let renderSlice = { nodes: this.sigmaInstance.graph.nodes(), edges: this.sigmaInstance.graph.edges() }
        let payload = Object.assign({}, this.graph, renderSlice);
        this.store.dispatch({ type: 'GRAPH_SET', payload: payload });
